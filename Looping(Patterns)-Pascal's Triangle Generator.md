## ðŸ§ª Program
```
def factorial(n): 

   if n == 0 or n == 1: 

      return 1 

   return n * factorial(n - 1) 

def combination(n, k): 

   return factorial(n) // (factorial(k) * factorial(n - k)) 

num_rows = int(input()) 

for i in range(num_rows):  
   print(' ' * (num_rows - i - 1), end='') 
   for j in range(i + 1): 
      print(combination(i, j), end=' ') 
   print() 
```
